üìã How to Use

Add this simple script tag to your website:

<!-- Basic Integration -->
<script 
  src="http://localhost:3000/widget-loader.js"
  data-nowgray-company-id="YOUR_COMPANY_ID"
  data-nowgray-widget-id="YOUR_WIDGET_ID"
  data-nowgray-position="bottom-right"
  data-nowgray-primary-color="#667eea"
  data-nowgray-secondary-color="#764ba2"
></script>

üìñ Configuration Options

// Manual initialization
<script src="http://localhost:3000/widget-loader.js"></script>
<script>
  initNowgrayChatWidget({
    companyId: 'YOUR_COMPANY_ID',      // Required
    widgetId: 'YOUR_WIDGET_ID',        // Optional
    position: 'bottom-right',          // bottom-right, bottom-left, top-right, top-left
    primaryColor: '#667eea',           // Primary brand color
    secondaryColor: '#764ba2',         // Secondary brand color
    apiUrl: 'http://localhost:5001',   // API endpoint
    fontFamily: 'Inter, sans-serif',   // Font family
    debug: true                        // Enable debug logging
  });
</script>

// Show/hide widget programmatically
